14// Adapted in 2010 for impulse response peak-minimization by convex iteration by
15// Christine Law.

17// "FIR Filter Design via Spectral Factorization and Convex Optimization"
18// by S.-P. Wu, S. Boyd, and L. Vandenberghe
19//
20// Designs an FIR lowpass filter using spectral factorization method with
21// constraint on maximum passband ripple and stopband attenuation:
22//
23// minimize max |H(w)| for w in stopband
24// s.t. 1/delta <= |H(w)| <= delta for w in passband
25//
26// We change variables via spectral factorization method and get:
27//
28// minimize max R(w) for w in stopband
29// s.t. (1/delta)**2 <= R(w) <= delta**2 for w in passband
30// R(w) >= 0 for all w
31//
32// where R(w) is squared magnitude frequency response
33// (and Fourier transform of autocorrelation coefficients r).
34// Variables are coeffients r and G = hh' where h is impulse response.
35// delta is allowed passband ripple.
36// This is a convex problem (can be formulated as an SDP after sampling).

126inline auto create_lowpass_case(size_t N = 48) -> std::pair<LowpassOracle, double> {
127 const auto delta0_wpass = 0.025;
128 const auto delta0_wstop = 0.125;
129 // maximum passband ripple in dB (+/- around 0 dB)
130 const auto delta1 = 20.0 * std::log10(1.0 + delta0_wpass);
131 // stopband attenuation desired in dB
132 const auto delta2 = 20.0 * std::log10(delta0_wstop);
133
134 // *********************************************************************
135 // optimization parameters
136 // *********************************************************************
137 // rule-of-thumb discretization (from Cheney's Approximation Theory)
138 const auto Lp = std::pow(10, -delta1 / 20);
139 const auto Up = std::pow(10, +delta1 / 20);
140 const auto Sp = std::pow(10, +delta2 / 20);
141
142 auto omega = LowpassOracle(N, Lp * Lp, Up * Up, 0.12, 0.20);
143 return {std::move(omega), Sp * Sp};
144}
