281 auto _update_core(Vec& grad, const T& beta, Fn&& cut_strategy) -> CutStatus {
282 std::valarray<double> grad_t(0.0, this->_n);
283 for (size_t i = 0; i != this->_n; ++i) {
284 for (size_t j = 0; j != this->_n; ++j) {
285 grad_t[i] += this->_mq(i, j) * grad[j];
286 }
287 }
288
289 const auto omega = (grad_t * grad).sum();
290 this->_tsq = this->_kappa * omega;
291
292 if (omega == 0.0) {
293 grad = grad_t;
294 return CutStatus::Success;
295 }
296
297 auto result = std::forward<Fn>(cut_strategy)(beta, this->_tsq);
298 if (result.status != CutStatus::Success) {
299 return result.status;
300 }
301
302 // n (n+1) / 2 + n
303 const auto r = result.sigma / omega;
304 for (size_t i = 0; i != this->_n; ++i) {
305 const auto rQg = r * grad_t[i];
306 for (size_t j = 0; j != i; ++j) {
307 this->_mq(i, j) -= rQg * grad_t[j];
308 this->_mq(j, i) = this->_mq(i, j);
309 }
310 this->_mq(i, i) -= rQg * grad_t[i];
311 }
312
313 this->_kappa *= result.delta;
314
315 if (this->no_defer_trick) {
316 this->_mq *= this->_kappa;
317 this->_kappa = 1.0;
318 }
319
320 grad = grad_t * (result.rho / omega);
321 return result.status;
322 }
