143inline Arr view(const Arr& a, const Range& rows, const Range& cols) {
144 assert(a.is_2d());
145 auto rs = rows.start;
146 auto re = (rows.end == Range::ALL) ? a.rows() : rows.end;
147 auto cs = cols.start;
148 auto ce = (cols.end == Range::ALL) ? a.cols() : cols.end;
149 auto rstep = rows.step;
150 auto cstep = cols.step;
151 size_t out_r = (re - rs + rstep - 1) / rstep;
152 size_t out_c = (ce - cs + cstep - 1) / cstep;
153 Arr out(out_r, out_c);
154 for (size_t i = 0; i < out_r; ++i)
155 for (size_t j = 0; j < out_c; ++j) out(i, j) = a(rs + i * rstep, cs + j * cstep);
156 return out;
157}
158
163inline Arr view(const Arr& a, const Range& rows) {
164 if (!a.is_2d()) {
165 auto s = rows.start;
166 auto e = (rows.end == Range::ALL) ? a.size() : rows.end;
167 auto step = rows.step;
168 size_t n = (e - s + step - 1) / step;
169 Arr out(n);
170 for (size_t i = 0; i < n; ++i) out(i) = a(s + i * step);
171 return out;
172 }
173 return view(a, rows, Range(Range::ALL));
174}

198inline Arr linspace(double start, double end, size_t n) {
199 Arr out(n);
200 if (n == 0) return out;
201 if (n == 1) {
202 out(0) = start;
203 return out;
204 }
205 double step = (end - start) / static_cast<double>(n - 1);
206 for (size_t i = 0; i < n; ++i) out(i) = start + step * static_cast<double>(i);
207 return out;

288inline Arr dot(const Arr& A, const Arr& x) {
289 assert(A.is_2d() && !x.is_2d() && A.cols() == x.size());
290 Arr out(A.rows());
291 for (size_t i = 0; i < A.rows(); ++i) {
292 double s = 0.0;
293 for (size_t j = 0; j < A.cols(); ++j) s += A(i, j) * x(j);
294 out(i) = s;
295 }
296 return out;
297}

322inline Arr concatenate(const Arr& a, const Arr& b, int /* axis */ = 1) {
323 assert(a.is_2d() && b.is_2d() && a.rows() == b.rows());
324 size_t m = a.rows();
325 size_t ca = a.cols();
326 size_t cb = b.cols();
327 Arr out(m, ca + cb);
328 for (size_t i = 0; i < m; ++i) {
329 for (size_t j = 0; j < ca; ++j) out(i, j) = a(i, j);
330 for (size_t j = 0; j < cb; ++j) out(i, ca + j) = b(i, j);
331 }
332 return out;
333}
