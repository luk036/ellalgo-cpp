288inline Vector0 conjugate_gradient(const Matrix0& A, const Vector0& b, const Vector0* x0 = nullptr,
289 double tol = 1e-5, int max_iter = 1000) {
290 size_t ndim = b.size();
291 Vector0 x_vector = x0 ? *x0 : Vector0(ndim);
292
293 Vector0 residual = b - A.dot(x_vector);
294 Vector0 director = residual;
295 double r_norm_sq = residual.dot(residual);
296
297 for (int i = 0; i < max_iter; ++i) {
298 Vector0 Ap = A.dot(director);
299 double alpha = r_norm_sq / director.dot(Ap);
300 x_vector += alpha * director;
301 residual -= alpha * Ap;
302 double r_norm_sq_new = residual.dot(residual);
303
304 if (std::sqrt(r_norm_sq_new) < tol) {
305 return x_vector;
306 }
307
308 double beta = r_norm_sq_new / r_norm_sq;
309 director = residual + beta * director;
310 r_norm_sq = r_norm_sq_new;
311 }
312
313 throw std::runtime_error("Conjugate Gradient did not converge after " + std::to_string(max_iter)
314 + " iterations");
315}
